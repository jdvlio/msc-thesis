\newglossaryentry{domain}{
	name={$\domain{k}$},
	sort={domain},
	description={For $k\in\posnats$, the map that takes an $L$-structure $\mathfrak{M}$ to the set $\domain{k}\mathfrak{M}=M^k$ of tuples $\bar{a}$ from the \textit{underlying set} $\domain{}\mathfrak{M}=M$ (of $\mathfrak{M}$) of length $k$.  In particular, if $\alpha=(A,<)$ is a linear order then $\domain{}\alpha=A$.  Additionally, as matter of convention, we choose $\domain{0}=\set{\emptyset}$.}
}
